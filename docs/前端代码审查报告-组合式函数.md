#### COMP-002: 缓存无过期机制

**位置**: `useDataLoader.ts:180-183`

**问题描述**:
```typescript
if (cacheEnabled) {
  cache.set(url, data.value)
}
```
- 缓存永不过期，可能导致数据更新不及时

**修复建议**:
```typescript
interface UseDataLoaderOptions<T> {
  // 添加缓存过期时间（毫秒）
  cacheTTL?: number
}

// 在load函数中
if (cacheEnabled && cache.has(url)) {
  const cached = cache.get(url)!
  // 检查缓存是否过期
  if (cached.timestamp && Date.now() - cached.timestamp < cacheTTL) {
    data.value = cached.data
    return
  }
  // 过期则删除
  cache.delete(url)
}

// 存入缓存时记录时间
if (cacheEnabled) {
  cache.set(url, {
    data: data.value,
    timestamp: Date.now()
  })
}
```

#### COMP-005: play()状态设置顺序错误

**位置**: `usePlaybackControl.ts:54-59`

**问题描述**:
```typescript
function play() {
  if (!audioRef.value) return

  audioRef.value.play().catch((err) => {
    console.error('播放失败:', err)
    isPlaying.value = false
  })
  isPlaying.value = true  // 在catch之前执行！
  options.onPlay?.()
}
```
- 如果`play()`返回的Promise reject，`isPlaying`已被设为true，然后才在catch中设为false
- 虽然最终结果正确，但有短暂的状态不一致

**修复建议**:
```typescript
async function play() {
  if (!audioRef.value) return

  try {
    await audioRef.value.play()
    isPlaying.value = true
    options.onPlay?.()
  } catch (err) {
    console.error('播放失败:', err)
    isPlaying.value = false
  }
}
```

**预估Bug率**: 10%（建议人工审核，因为修改涉及async）
