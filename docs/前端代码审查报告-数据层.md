#### AUTH-001: JWT手动解码安全风险

**位置**: `auth.ts:170-173`

**问题描述**:
```typescript
const tokenParts = token.value!.split('.')
if (tokenParts.length < 2) return true
const payload = JSON.parse(atob(tokenParts[1] as string))
```
- 使用`atob()`解码可能被恶意token利用
- 没有验证token签名的完整性

**修复建议**:
```typescript
import jwt from 'jsonwebtoken'

function isTokenExpired(): boolean {
  if (!token.value) return true
  
  try {
    const decoded = jwt.decode(token.value, { complete: true })
    if (!decoded || typeof decoded.payload !== 'object') return true
    
    const expiry = decoded.payload.exp
    if (!expiry) return true
    
    return Date.now() > expiry * 1000
  } catch {
    return true
  }
}
```

**预估Bug率**: 15-20%（需引入外部库，建议人工审核）

#### STUD-003: restoreFromStorage验证逻辑重复

**位置**: `student.ts:56-61`

**问题描述**:
验证逻辑与setStudentId可能不一致

**修复建议**:
```typescript
// 提取验证函数
function isValidStudentId(id: string): boolean {
  return id && /^\d{4}$/.test(id)
}

function setStudentId(id: string) {
  if (!isValidStudentId(id)) {
    console.warn(`[StudentStore] 无效的学号格式: ${id}`)
    return
  }
  studentId.value = id
  localStorage.setItem('studentId', id)
}

function restoreFromStorage() {
  const saved = localStorage.getItem('studentId')
  if (isValidStudentId(saved)) {
    studentId.value = saved
  }
}
```

**预估Bug率**: <5%（可安全修复）
