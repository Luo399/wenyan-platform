#### SERV-001: getLocalData无超时处理

**位置**: `apiService.ts:283-290`

**问题描述**:
```typescript
export async function getLocalData<T>(dataType: string, textId: string): Promise<T> {
  const path = getLocalJsonPath(dataType, textId)
  const response = await fetch(path)  // 无超时控制！
  if (!response.ok) {
    throw new Error(`无法加载本地数据: ${response.status} ${path}`)
  }
  return await response.json()
}
```
- fetch请求无超时控制，网络异常时可能无限等待
- 没有AbortController取消机制

**修复建议**:
```typescript
export async function getLocalData<T>(
  dataType: string, 
  textId: string, 
  timeout = 5000
): Promise<T> {
  const path = getLocalJsonPath(dataType, textId)
  const controller = new AbortController()
  const timeoutId = setTimeout(() => controller.abort(), timeout)
  
  try {
    const response = await fetch(path, { signal: controller.signal })
    clearTimeout(timeoutId)
    
    if (!response.ok) {
      throw new Error(`无法加载本地数据: ${response.status} ${path}`)
    }
    
    return await response.json()
  } catch (err) {
    clearTimeout(timeoutId)
    if (err instanceof DOMException && err.name === 'AbortError') {
      throw new Error(`加载本地数据超时: ${path}`)
    }
    throw err
  }
}
```

**预估Bug率**: 15-20%（可安全修复）

```typescript
// 添加缓存机制
const cache = new Map<string, Promise<any>>()

export async function getTextBasicInfo(textId: string): Promise<ApiResponse<TextBasicInfo>> {
  const key = `basic-info:${textId}`
  if (cache.has(key)) {
    return cache.get(key)!
  }
  
  const promise = get<TextBasicInfo>(`/api/texts/${textId}/basic-info`)
  cache.set(key, promise)
  
  // 缓存有效期5分钟
  setTimeout(() => cache.delete(key), 5 * 60 * 1000)
  
  return promise
}
```

### 8.2 添加请求重试

```typescript
// 添加重试机制
async function fetchWithRetry<T>(
  url: string, 
  options: RequestInit = {}, 
  retries = 2
): Promise<T> {
  try {
    const response = await fetch(url, options)
    if (!response.ok && retries > 0) {
      return fetchWithRetry(url, options, retries - 1)
    }
    return await response.json()
  } catch (err) {
    if (retries > 0) {
      return fetchWithRetry(url, options, retries - 1)
    }
    throw err
  }
}
```
